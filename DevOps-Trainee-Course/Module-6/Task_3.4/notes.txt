-------------------------Install Helm Chart------------------------
helm install \
  cert-manager oci://quay.io/jetstack/charts/cert-manager \
  --version v1.19.1 \
  --namespace cert-manager \
  --create-namespace \
  --set crds.enabled=true
-------------------------------------------------------------------

------------------------------------Install Nginx Ingress Controller-------------------------------------------------------------------
helm install ingress-nginx ingress-nginx/ingress-nginx \\n  -f nginx-values.yaml \\n  --namespace ingress-nginx \\n  --create-namespace
